import { Component, output, input, ChangeDetectionStrategy } from '@angular/core';
import { MatButtonModule } from '@angular/material/button';
import { MatBadgeModule } from '@angular/material/badge';
import { MatIconModule } from '@angular/material/icon';
import { MatTooltipModule } from '@angular/material/tooltip';
import { NgOptimizedImage, NgClass } from '@angular/common';

/**
 * A reusable Material Design mini floating action button (mini-FAB) component.
 * Features include:
 * - Compact circular button design
 * - Icon or image content
 * - Optional badge notification
 * - Tooltip support
 * - Theming options
 * - Click event emission
 */
@Component({
  selector: 'lib-mat-mini-fab-button',
  standalone: true,
  imports: [
    MatButtonModule,
    MatBadgeModule,
    MatIconModule,
    MatTooltipModule,
    NgOptimizedImage,
    NgClass
  ],
  template: `
    <button
      mat-mini-fab
      [color]="color()"
      [disabled]="disabled()"
      [type]="type()"
      [class]="(className() || '') + ' lib-mat-mini-fab-button'"
      [ngClass]="ngClassNames()"
      (click)="onClick()"
      [matBadge]="matBadge()"
      [matTooltip]="tooltip() || ''"
      [matTooltipPosition]="tooltipPosition()"
      [matTooltipShowDelay]="tooltipShowDelay()"
      [matTooltipHideDelay]="tooltipHideDelay()"
      [matTooltipDisabled]="tooltipDisabled() || !tooltip()"
    >
      @if (sourceType() == 'icon' && source()) {
        <mat-icon>{{ source() }}</mat-icon>
      }
      @else if (sourceType() == 'image' && source()) {
        <img [ngSrc]="source() ?? ''" width="20" height="20" alt="Loading..." />
      }
    </button>
  `,
  styles: [`
    :host {
      display: inline-block;
    }
    .lib-mat-mini-fab-button {
      transition: transform 0.2s ease, box-shadow 0.2s ease;
    }
    .lib-mat-mini-fab-button:hover {
      transform: scale(1.1);
    }
    .lib-mat-mini-fab-button:active {
      transform: scale(0.95);
    }
    .custom-icon {
      font-size: 20px;
      width: 20px;
      height: 20px;
    }
    .custom-image {
      border-radius: 50%;
    }
  `],
  changeDetection: ChangeDetectionStrategy.OnPush
})
export class LibMatMiniFabButtonComponent {
  /**
   * The color palette of the mini-FAB button.
   * Options include 'primary', 'accent', 'warn', or a custom color string.
   * @default 'primary'
   */
  readonly color = input<'primary' | 'accent' | 'warn' | string>('primary');

  /**
   * Content source type - either Material icon or custom image.
   * @default 'icon'
   */
  readonly sourceType = input<'icon' | 'image'>('icon');

  /**
   * Disabled state of the button.
   * @default false
   */
  readonly disabled = input<boolean>(false);

  /**
   * Native HTML button type attribute.
   * @default 'button'
   */
  readonly type = input<'button' | 'submit' | 'reset'>('button');

  /**
   * Additional CSS classes for custom styling.
   * @default null
   */
  readonly className = input<string | null>(null);

  readonly ngClassNames = input<string | string[] | Set<string> | {[klass: string]: any} | null | undefined>(undefined);

  /**
   * Badge content displayed in the upper right corner.
   * @default null
   */
  readonly matBadge = input<string | null>(null);

  /**
   * Icon name (for 'icon' sourceType) or image URL (for 'image' sourceType).
   * @default null
   */
  readonly source = input<string | null>(null);

  /**
   * Tooltip text displayed on hover.
   * @default null
   */
  readonly tooltip = input<string | null>(null);

  /**
   * Position of the tooltip relative to the button.
   * @default 'above'
   */
  readonly tooltipPosition = input<'above' | 'below' | 'left' | 'right'>('above');

  /**
   * Delay in ms before showing the tooltip.
   * @default 200
   */
  readonly tooltipShowDelay = input<number>(200);

  /**
   * Delay in ms before hiding the tooltip.
   * @default 0
   */
  readonly tooltipHideDelay = input<number>(0);

  /**
   * Whether to disable the tooltip.
   * @default false
   */
  readonly tooltipDisabled = input<boolean>(false);

  /**
   * Event emitted when the button is clicked.
   */
  matClick = output<void>();

  /**
   * Handles click events and emits the matClick output.
   */
  onClick(): void {
    this.matClick.emit();
  }
}
