  ChangeDetectionStrategy,
  Component,
  input,
  output,
  signal,
  effect,

@Component({
  selector: 'lib-file-picker',
  standalone: true,
  imports: [CommonModule, MatButtonModule, ReactiveFormsModule],
  changeDetection: ChangeDetectionStrategy.OnPush,
  template: `
      @if (label() && labelPosition() === 'OutSide') {
        <label class="lib-mat-label-outside">
          {{ label() }}
          @if (control().hasValidator(validatorRequired) && showAsterisk()) {
            <span class="lib-mat-label-outside-required" aria-hidden="true">*</span>
          }
        </label>
      }

    <div
      class="picker"
      [class.dragging]="dragging()"
      (dragover)="onDragOver($event)"
      (dragleave)="onDragLeave()"
      (drop)="onDrop($event)"
    >
      <input
        hidden
        #fileInput
        type="file"
        [accept]="accept()"
        [multiple]="multiple()"
        (change)="onSelect($event)"
      />

      <button mat-stroked-button type="button" (click)="fileInput.click()">
        Select File{{ multiple() ? 's' : '' }}
      </button>

      @if (enableDrop()) {
      <span class="hint">or drag & drop here</span>
      }
    </div>

    @if (preview() && files().length) {
    <div class="preview-grid">
      @for (item of files(); track item.file.name) {
      <div class="preview-card">
        @if (isImage(item.file) && imagePreview()) {
        <img [src]="item.preview" />
        } @else {
        <div class="file-icon">📄</div>
        }

        <div class="meta">
          <span class="name">{{ item.file.name }}</span>
          <small>{{ formatSize(item.file.size) }}</small>
        </div>

        <button type="button" class="remove" (click)="remove(item)">×</button>
      </div>
      }
    </div>
    } @if (control().invalid && (control().touched || control().dirty)) {
    <div class="error">Invalid file selection</div>
    }
  `,

export class LibFilePickerComponent {
  /** Inputs */
  label = input('');
  multiple = input(false);
  accept = input('*/*');
  preview = input(true);
  imagePreview = input(true);
  enableDrop = input(true);
  maxFileSizeMB = input(10);
  maxFiles = input<number | null>(null); // ✅ New input for max files
  showAsterisk = input<boolean>(true);
  labelPosition = input<'Inside' | 'OutSide'>('Inside');

  control = input.required<FormControl<File | File[] | null>>();
  upload = output<File[]>();

  dragging = signal(false);
  files = signal<PickerFile[]>([]);
  fileError = signal<string | null>(null);

  protected validatorRequired = Validators.required;

  onSelect(event: Event) {
    const input = event.target as HTMLInputElement;
    if (!input.files) return;
    this.handleFiles(Array.from(input.files));
    input.value = '';
  }

  onDrop(event: DragEvent) {
    event.preventDefault();
    this.dragging.set(false);
    if (!event.dataTransfer?.files) return;
    this.handleFiles(Array.from(event.dataTransfer.files));
  }

  onDragOver(event: DragEvent) {
    event.preventDefault();
    this.dragging.set(true);
  }

  onDragLeave() {
    this.dragging.set(false);
  }

  handleFiles(files: File[]) {
    // ✅ Check max file count
    if (this.maxFiles() && this.files().length + files.length > (this.maxFiles() ?? 0)) {
      this.fileError.set(`You can select maximum ${this.maxFiles()} file(s).`);
      return;
    }

    // ✅ Filter by max file size
    const validFiles = files.filter(f => f.size / 1024 / 1024 <= this.maxFileSizeMB());
    if (!validFiles.length) {
      this.fileError.set(`File(s) exceed max size of ${this.maxFileSizeMB()} MB.`);
      return;
    }

    // Clear previous error
    this.fileError.set(null);

    validFiles.forEach(file => {
      this.createPreview(file).then(preview => {
        const entry: PickerFile = { file, preview, progress: null };
        this.files.set(this.multiple() ? [...this.files(), entry] : [entry]);
        this.control().setValue(
          this.multiple() ? this.files().map(f => f.file) : entry.file
        );
      });
    });

    this.upload.emit(validFiles);
  }

  private createPreview(file: File): Promise<string | undefined> {
    if (!this.isImage(file)) return Promise.resolve(undefined);
    return new Promise(resolve => {
      const reader = new FileReader();
      reader.onload = () => resolve(reader.result as string);
      reader.readAsDataURL(file);
    });
  }

  isImage(file: File): boolean {
    return file.type.startsWith('image/');
  }

  remove(item: PickerFile) {
    this.files.update(f => f.filter(x => x !== item));
    this.control().setValue(
      this.multiple() ? this.files().map(f => f.file) : null
    );
    this.fileError.set(null); // Clear error on remove
  }

  formatSize(bytes: number): string {
    return (bytes / 1024 / 1024).toFixed(1) + ' MB';
  }
}

type PickerFile = {
  file: File;
  preview?: string;
  progress: number | null;
};
