export class LibDialogComponent implements AfterViewInit {
  /**
   * Reference to the dynamic content container within the dialog.
   */
  private $dialogContent = viewChild('dialogContent', { read: ViewContainerRef });

  /**
   * Reference to the dynamic template component datay
   * .
   */
  protected $componentData = inject<IDialogData>(MAT_DIALOG_DATA);

  /**
   * Reference to the dynamically created component.
   */
  private componentRef = signal<ComponentRef<any> | null>(null);

  /**
   * Injected Angular Material dialog reference to control dialog actions.
   */
  private $dialogRef = inject(MatDialogRef<LibDialogComponent>);

  protected resolveDynamic = resolveDynamic;

  /**
   * Lifecycle hook that runs after the view has been initialized.
   * Loads the dynamic component, if any.
   */
  ngAfterViewInit(): void {
    this.loadComponent();
  }

  /**
   * Dynamically loads a component inside the dialog.
   */
  private loadComponent(): void {
    if (this.$componentData.component) {
      const containerRef = this.$dialogContent()!;
      containerRef.clear();
      if (this.$componentData.component) {
        this.componentRef.set(containerRef.createComponent(resolveDynamic(this.$componentData.component)));
        if (this.$componentData.componentData) {
          Object.assign(this.componentRef()?.instance, this.$componentData.componentData);
        }
      }
    }
  }

  /**
   * Closes the dialog and returns `false` as the result.
   */
  onCancel(): void {
    this.$dialogRef.close(false);
  }

  /**
   * Closes the dialog and returns `true` as the result.
   */
  onConfirm(): void {
    this.$dialogRef.close(true);
  }
}
